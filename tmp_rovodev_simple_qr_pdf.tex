\section{Step 1: Install Zsh}

\subsection{Ubuntu/Debian Systems}

\codeqr{Update Package List}{sudo apt update}

\codeqr{Install Zsh}{sudo apt install zsh}

\subsection{CentOS/RHEL/Fedora Systems}

\codeqr{Install Zsh (CentOS/RHEL)}{sudo yum install zsh}

\codeqr{Install Zsh (Fedora)}{sudo dnf install zsh}

\subsection{macOS Systems}

\codeqr{Install Zsh (Homebrew)}{brew install zsh}

\subsection{Verify Installation}

\codeqr{Check Zsh Version}{zsh --version}

\codeqr{Find Zsh Location}{which zsh}

\section{Step 2: Install Oh My Zsh}

\subsection{Method 1: Using curl}

\codeqr{Install Oh My Zsh (curl)}{sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"}

\subsection{Method 2: Using wget}

\codeqr{Install Oh My Zsh (wget)}{sh -c "$(wget https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh -O -)"}

\subsection{Method 3: Manual Installation}

\codeqr{Clone Oh My Zsh Repository}{git clone https://github.com/ohmyzsh/ohmyzsh.git ~/.oh-my-zsh}

\codeqr{Copy Template Configuration}{cp ~/.oh-my-zsh/templates/zshrc.zsh-template ~/.zshrc}

\section{Step 3: Change Default Shell}

\subsection{Standard Systems}

\codeqr{Change Shell (Standard)}{chsh -s $(which zsh)}

\subsection{GitHub Codespaces}

\codeqr{Change Shell (Codespaces Method 1)}{sudo chsh -s $(which zsh) $(whoami)}

\codeqr{Change Shell (Codespaces Method 2)}{sudo chsh "$(id -un)" --shell "$(which zsh)"}

\section{Step 4: Install Popular Plugins}

\subsection{Zsh Autosuggestions}

\codeqr{Install Autosuggestions Plugin}{git clone https://github.com/zsh-users/zsh-autosuggestions ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-autosuggestions}

\subsection{Zsh Syntax Highlighting}

\codeqr{Install Syntax Highlighting Plugin}{git clone https://github.com/zsh-users/zsh-syntax-highlighting.git ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-syntax-highlighting}

\subsection{Powerlevel10k Theme}

\codeqr{Install Powerlevel10k Theme}{git clone --depth=1 https://github.com/romkatv/powerlevel10k.git ${ZSH_CUSTOM:-$HOME/.oh-my-zsh/custom}/themes/powerlevel10k}

\section{Step 5: Apply Changes}

\codeqr{Reload Configuration}{source ~/.zshrc}

\section{Verification Commands}

\codeqr{Check Current Shell}{echo $SHELL}

\codeqr{Check User Shell Setting}{grep $(whoami) /etc/passwd}

\section{Troubleshooting}

\codeqr{Fix Permissions (if needed)}{sudo chsh -s $(which zsh) $USER}

\codeqr{Check Plugin Directory}{ls ~/.oh-my-zsh/custom/plugins/}

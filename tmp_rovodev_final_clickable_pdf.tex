\section{Step 3: Change Default Shell}

\subsection{Standard Systems}

\commandbox{Change Shell (Standard)}{chsh -s $(which zsh)}{chsh -s \$(which zsh)}

\subsection{GitHub Codespaces}

\commandbox{Change Shell (Codespaces Method 1)}{sudo chsh -s $(which zsh) $(whoami)}{sudo chsh -s \$(which zsh) \$(whoami)}

\commandbox{Change Shell (Codespaces Method 2)}{sudo chsh "$(id -un)" --shell "$(which zsh)"}{sudo chsh "\$(id -un)" --shell "\$(which zsh)"}
